programa sumatoria;
vars n, res: entero;
nula calc_sumatoria(n: entero) {
    si(n == 0){
        res = 0;
    } sino {
        calc_sumatoria(n - 1);
        res = res + n;
    };
};
inicio {
    escribe("--- TEST SUMATORIA ---");
    n = 10;
    calc_sumatoria(n);
    escribe("Sumatoria hasta n:", res);
} fin
